SQL code used for analysis:

SELECT b.name, 
        CASE 
            WHEN b.stars between 2 AND 3 THEN "2-3"
            WHEN b.stars between 4 AND 5 THEN "4-5"
        END AS star_rating,
        GROUP_CONCAT(h.hours),
        b.review_count,
        b.postal_code,
        b.neighborhood,
        b.longitude,
        b.latitude
FROM business b
JOIN category c ON b.id = c.business_id
JOIN hours h ON b.id = h.business_id
WHERE 
    b.city IS "Phoenix" 
    AND c.category IS "Restaurants"
    AND (star_rating IS "2-3" OR star_rating IS "4-5")
GROUP BY b.name, star_rating
ORDER BY star_rating DESC;

SQL code used for analysis:

SELECT 
    b.is_open, 
    COUNT(b.id) num_businesses,
    ROUND(AVG(stars), 2) avg_star_rating, 
    ROUND(AVG(review_count), 2) avg_review_count

iv. Provide the SQL code you used to create your final dataset:

SELECT 
    b.name,
    c.category,
    b.city,
    b.state,
    AVG(r.stars) avg_stars_rating,
    AVG(r.useful) avg_useful_votes,
    AVG(r.funny) avg_funny_votes,
    AVG(r.cool) avg_cool_votes,
    GROUP_CONCAT(r.text) review_text
FROM business b 
LEFT JOIN category c ON b.id = c.business_id
INNER JOIN review r ON b.id = r.business_id
GROUP BY b.name, b.city, b.state
ORDER BY b.name, b.city, b.state;
